remove_outliers<- function(a) {
    sdv <- sd(a)
    q1<-quantile(a,0.25)
    q3<-quantile(a,0.75)
    H<-1.5*IQR(a)
    min <- min(a)
    max <- max(a)
    UB <- q3+H
    LB <-q1-H 
    outlier_flag<- max>UB | min<LB
    return(c(outlier_flag=outlier_flag,min = min,q1=q1,q3=q3,max=max, UB=UB, LB=LB ))
}
